// component.ts
activeIdx = 0;
leftIdx = 2;
rightIdx = 1;
progressKey = 0;

carousel = [
  { img: 'assets/best-seller/best-2.png', alt: 'Model A' },
  { img: 'assets/best-seller/best-3.png', alt: 'Model B' },
  { img: 'assets/best-seller/best-1.png', alt: 'Model C' },
];

// double buffer sources
aSrc = this.carousel[0].img;   // top layer at start
bSrc = this.carousel[1].img;   // back layer preload
frontIsA = true;

private rotTimer: any;
private readonly slideMs = 3000;   // 3s
private readonly xfadeMs = 500;    // CSS duration must match

ngOnInit() {
  this.computeNeighbors();
  this.start();
}

ngOnDestroy() { this.pause(); }

private computeNeighbors() {
  const n = this.carousel.length;
  this.leftIdx  = (this.activeIdx + n - 1) % n;
  this.rightIdx = (this.activeIdx + 1) % n;
}

// start / pause for hover
start() { this.progressKey = this.activeIdx; this.rotTimer = setInterval(() => this.next(), this.slideMs); }
pause() { if (this.rotTimer) { clearInterval(this.rotTimer); this.rotTimer = null; } }
resume() { if (!this.rotTimer) this.start(); }

private next() {
  const nextIdx = (this.activeIdx + 1) % this.carousel.length;

  // preload next on the back layer, then flip for silky transition
  if (this.frontIsA) {
    this.bSrc = this.carousel[nextIdx].img;
    // allow DOM to paint before flipping
    setTimeout(() => this.frontIsA = false, 20);
  } else {
    this.aSrc = this.carousel[nextIdx].img;
    setTimeout(() => this.frontIsA = true, 20);
  }

  // update indexes after crossfade
  setTimeout(() => {
    this.activeIdx = nextIdx;
    this.computeNeighbors();
    this.progressKey = this.activeIdx; // restart progress animation
  }, this.xfadeMs);
}
